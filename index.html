Connect to Supabase

Open this file in a text editor and replace the two placeholder values in the SUPABASE CONFIGURATION section near the top of the <script>. You also need to run the SQL below in Supabase's SQL Editor to create the two required tables.

create table campaigns (
  id bigint generated always as identity primary key,
  created_at timestamptz default now(),
  name text not null unique,
  is_active boolean default true
);

create table survey_responses (
  id bigint generated always as identity primary key,
  created_at timestamptz default now(),
  campaign_id bigint references campaigns(id),
  campaign_name text,
  team text not null,
  avg_score numeric not null,
  dim_avgs jsonb,
  answers jsonb,
  comment text,
  browser_id text
);

alter table campaigns enable row level security;
alter table survey_responses enable row level security;

create policy "public insert campaigns" on campaigns for insert with check (true);
create policy "public select campaigns" on campaigns for select using (true);
create policy "public update campaigns" on campaigns for update using (true);
create policy "public delete campaigns" on campaigns for delete using (true);
create policy "public insert responses" on survey_responses for insert with check (true);
create policy "public select responses" on survey_responses for select using (true);
create policy "public delete responses" on survey_responses for delete using (true);
